using System;

namespace BedrockSeedFinder
{
    public static class StrongholdFinder
    {
        // Stronghold constants
        private const int STRONGHOLD_COUNT = 3;
        private const int STRONGHOLD_GRID_SIZE = 200;

        public static bool CheckStronghold(int worldSeed, int targetX, int targetZ, int distance)
        {
            BedrockRandom mt = new BedrockRandom(worldSeed);
            float angle = mt.NextFloat() * (float)Math.PI * 2.0f;
            int radius = mt.NextInt(16) + 40;

            for (int i = 0; i < STRONGHOLD_COUNT; i++)
            {
                int cx = (int)Math.Floor(radius * Math.Cos(angle));
                int cz = (int)Math.Floor(radius * Math.Sin(angle));

                bool found = false;
                for (int x = cx - 8; x < cx + 8 && !found; x++)
                {
                    for (int z = cz - 8; z < cz + 8 && !found; z++)
                    {
                        if (VillageFinder.CheckVillageAtChunk(worldSeed, x, z))
                        {
                            int blockX = x << 4;
                            int blockZ = z << 4;

                            int dx = blockX - targetX;
                            int dz = blockZ - targetZ;
                            double dist = Math.Sqrt(dx * dx + dz * dz);

                            if (dist <= distance)
                                return true;

                            found = true;
                        }
                    }
                }
                angle += (found ? 0.6f : 0.25f) * (float)Math.PI;
                radius += (found ? 8 : 4);
            }

            int targetRegX = (int)Math.Floor((double)targetX / (STRONGHOLD_GRID_SIZE * 16));
            int targetRegZ = (int)Math.Floor((double)targetZ / (STRONGHOLD_GRID_SIZE * 16));
            int regionRadius = (int)Math.Ceiling((double)distance / (STRONGHOLD_GRID_SIZE * 16)) + 2;

            for (int regX = targetRegX - regionRadius; regX <= targetRegX + regionRadius; regX++)
            {
                for (int regZ = targetRegZ - regionRadius; regZ <= targetRegZ + regionRadius; regZ++)
                {
                    int regX100 = STRONGHOLD_GRID_SIZE * regX + 100;
                    int regZ100 = STRONGHOLD_GRID_SIZE * regZ + 100;
                    long xMul = (long)-1683231919 * regX100;
                    long zMul = (long)-1100435783 * regZ100;
                    long seed = xMul + zMul + worldSeed + 97858791;
                    BedrockRandom regionMt = new BedrockRandom((int)seed);

                    if (regionMt.NextFloat() < 0.25f)
                    {
                        int minX = STRONGHOLD_GRID_SIZE * regX + STRONGHOLD_GRID_SIZE - 150;
                        int minZ = STRONGHOLD_GRID_SIZE * regZ + STRONGHOLD_GRID_SIZE - 150;
                        int maxX = STRONGHOLD_GRID_SIZE * regX + 150;
                        int maxZ = STRONGHOLD_GRID_SIZE * regZ + 150;

                        int x = regionMt.NextInt(minX, maxX);
                        int z = regionMt.NextInt(minZ, maxZ);

                        int strongholdX = x << 4;
                        int strongholdZ = z << 4;

                        int dx = strongholdX - targetX;
                        int dz = strongholdZ - targetZ;
                        double dist = Math.Sqrt(dx * dx + dz * dz);

                        if (dist <= distance)
                            return true;
                    }
                }
            }
            return false;
        }
    }
}
